17  package org.apache.commons.rng.sampling.distribution;
18  
19  /**
20   * Sampling from a Gaussian distribution with given mean and
21   * standard deviation.
22   *
23   * @since 1.1
24   */
25  public class GaussianSampler implements ContinuousSampler {
26      /** Mean. */
27      private final double mean;
28      /** standardDeviation. */
29      private final double standardDeviation;
30      /** Normalized Gaussian sampler. */
31      private final NormalizedGaussianSampler normalized;
32  
33      /**
34       * @param normalized Generator of N(0,1) Gaussian distributed random numbers.
35       * @param mean Mean of the Gaussian distribution.
36       * @param standardDeviation Standard deviation of the Gaussian distribution.
37       */
38      public GaussianSampler(NormalizedGaussianSampler normalized,
39                             double mean,
40                             double standardDeviation) {
41          this.normalized = normalized;
42          this.mean = mean;
43          this.standardDeviation = standardDeviation;
44      }
45  
46      /** {@inheritDoc} */
47      @Override
48      public double sample() {
49          return standardDeviation * normalized.sample() + mean;
50      }
51  
52      /** {@inheritDoc} */
53      @Override
54      public String toString() {
55          return "Gaussian deviate [" + normalized.toString() + "]";
56      }
57  }
