17  package org.apache.commons.rng;
18  
19  /**
20   * Applies to generators whose internal state can be saved and restored.
21   *
22   * @since 1.0
23   */
24  public interface RestorableUniformRandomProvider extends UniformRandomProvider {
25      /**
26       * Saves the state of a generator.
27       *
28       * @return the current state of this instance. It is a value that can
29       * subsequently be passed to the {@link #restoreState(RandomProviderState)
30       * restore} method.
31       * @throws UnsupportedOperationException if the underlying source of
32       * randomness does not support this functionality.
33       */
34      RandomProviderState saveState();
35  
36      /**
37       * Restores the state of a generator.
38       *
39       * @param state State which this instance will be set to.
40       * This parameter would usually have been obtained by a call to
41       * {@link #saveState() saveState} performed either on the same
42       * object as this one, or an object of the exact same class.
43       * @throws UnsupportedOperationException if the underlying source of
44       * randomness does not support this functionality.
45       * @throws IllegalArgumentException if it was detected that the
46       * {@code state} argument is incompatible with this intance.
47       */
48      void restoreState(RandomProviderState state);
49  }
